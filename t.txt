const puppeteer = require('puppeteer');

async function generatePDF(htmlContent) {
  const browser = await puppeteer.launch({
    headless: true,
    args: ['--no-sandbox', '--disable-setuid-sandbox']
  });
  
  const page = await browser.newPage();
  await page.setContent(htmlContent, { waitUntil: 'domcontentloaded' });
  
  const pdfBuffer = await page.pdf({
    format: 'A4',
    printBackground: true,
    margin: {
      top: '10mm',
      bottom: '10mm',
      left: '10mm',
      right: '10mm'
    }
  });

  await page.close();
  await browser.close();
  
  return Buffer.from(pdfBuffer);
}

const nodemailer = require('nodemailer');

async function sendReportEmail(toEmail, subject, htmlBody, pdfBuffer, filename) {
  // 1. שליפת הגדרות SMTP מה-DB
  const settings = await db.select().from(emailSettings).limit(1);
  
  if (!settings || !settings.isActive) {
    throw new Error('הגדרות מייל לא מוגדרות');
  }
  
  // 2. יצירת transporter
  const transporter = nodemailer.createTransport({
    host: settings.smtpHost,
    port: settings.smtpPort,
    secure: settings.smtpSecure,  // true לפורט 465
    auth: {
      user: settings.smtpUser,
      pass: settings.smtpPassword,
    },
  });
  
  // 3. אימות החיבור
  await transporter.verify();
  
  // 4. שליחת המייל
  await transporter.sendMail({
    from: `"${settings.fromName}" <${settings.fromEmail}>`,
    to: toEmail,
    subject: subject,
    html: htmlBody,
    attachments: [{
      filename: filename,
      content: pdfBuffer
    }]
  });
}

// GET /api/reports/send-bulk-stream?month=1&year=2026
// מחזיר SSE (Server-Sent Events) להתקדמות בזמן אמת

app.get("/api/reports/send-bulk-stream", async (req, res) => {
  const { month, year } = req.query;
  
  // הגדרת SSE
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  
  const sendEvent = (event, data) => {
    res.write(`event: ${event}\ndata: ${JSON.stringify(data)}\n\n`);
  };
  
  // שליפת כל המדריכים עם מייל
  const guides = await db.select().from(people)
    .where(and(
      eq(people.isActive, true),
      eq(roles.name, 'guide')
    ));
  
  sendEvent('start', { total: guides.length });
  
  let sent = 0, skipped = 0;
  
  for (const guide of guides) {
    // דילוג אם אין מייל
    if (!guide.email) {
      skipped++;
      sendEvent('progress', { current: guide.name, status: 'skipped', reason: 'אין מייל' });
      continue;
    }
    
    try {
      // יצירת ושליחת הדוח
      const reportData = await getGuideReportData(guide.id, month, year);
      
      if (reportData.rows.length === 0) {
        skipped++;
        sendEvent('progress', { current: guide.name, status: 'skipped', reason: 'אין משמרות' });
        continue;
      }
      
      const html = generateReportHTML(reportData);
      const pdf = await generatePDF(html);
      await sendReportEmail(guide.email, ...);
      
      sent++;
      sendEvent('progress', { current: guide.name, status: 'sent', sent, skipped });
      
    } catch (error) {
      skipped++;
      sendEvent('progress', { current: guide.name, status: 'failed', reason: error.message });
    }
  }
  
  sendEvent('complete', { sent, skipped });
  res.end();
});

const handleSendBulk = () => {

  );
  eventSourceRef.current = eventSource;

  // אירוע: התחלה
  eventSource.addEventListener('start', (event) => {

  // אירוע: התקדמות
  eventSource.addEventListener('progress', (event) => {
    const data = JSON.parse(event.data);

    //   processed: 23, total: 45, 
    //   current: "דוד משה", currentId: 123,
    //   status: "sent" | "skipped", reason?: "אין משמרות",

      current: data.current,
      sent: data.sent,
      skipped: data.skipped
    });
  });

  // אירוע: סיום
  eventSource.addEventListener('complete', (event) => {
    const data = JSON.parse(event.data);
    // data = {

const ProgressBar = ({ progress }: { progress: ProgressState }) => {
